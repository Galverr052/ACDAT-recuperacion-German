 @GetMapping(value = "/busca", produces = MediaType.APPLICATION_JSON_VALUE)
    @ResponseBody
    public Persona getPersonaNom(@RequestParam String nombre) {
        boolean encontrado = false;
        Persona persona = null;

        Iterator it = personas.iterator();

        while (!encontrado && it.hasNext()) {
            persona = (Persona) it.next();

            if (persona.getNombre().equals(nombre))
                encontrado = true;
        }

				//return encontrado ? persona: null;

        if (encontrado)
            return persona;
        else
            return null;
    }

@GetMapping(value = "/busca", produces = MediaType.APPLICATION_JSON_VALUE)
    @ResponseBody
    public Persona buscaPersona(@RequestParam String nombre) {
        return personas.stream()
                .filter(p -> p.getNombre().equalsIgnoreCase(nombre))
                .findFirst()
                .orElse(null);
    }

            return "El parámetro minimo debe ser menor o igual al máximo";

@GetMapping(value = "ex/nveces",  produces = "application/xml")
    @ResponseBody
    public String frase(@RequestParam String frase) {

        StringBuilder xml = new StringBuilder();
        xml.append("<lista>");

        String procesados = "";

        for (char c : frase.toCharArray()) {

            if (procesados.indexOf(c) == -1) {
                int count = 0;

                for (char x : frase.toCharArray()) {
                    if (x == c) count++;
                }

                xml.append("<caracter>");
                xml.append("<simbolo>").append(c).append("</simbolo>");
                xml.append("<nveces>").append(count).append("</nveces>");
                xml.append("</caracter>");

                procesados += c;
            }
        }

        xml.append("</lista>");
        return xml.toString();
    }

// <server>/ws/ex/nvecesCaract?frase=a3$3Ab
    @GetMapping(value = "/ex/nvecesCaract", produces = MediaType.APPLICATION_XML_VALUE)
    @ResponseBody
    public List<Caracter> contar(@RequestParam String frase) {

        Map<Character, Integer> contador = new LinkedHashMap<>(); //linkedHashMap porque guarda el orden de inserción a diferencia de HashMap

        // Contar caracteres
        for (int i = 0; i < frase.length(); i++) {
            char c = frase.charAt(i);

                if (contador.containsKey(c)) {
                    contador.put(c, contador.get(c) + 1);
                } else {
                    contador.put(c, 1);
                }
        }

        // Paso de Map a List
        List<Caracter> resultado = new ArrayList<>();

        for (Map.Entry<Character, Integer> entry : contador.entrySet()) {
            resultado.add(
                    new Caracter(
                            String.valueOf(entry.getKey()),
                            entry.getValue()
                    )
            );
        }

        return resultado;
    }
